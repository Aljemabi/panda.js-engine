Object pooling

game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function() {

SceneGame = game.Scene.extend({
    init: function() {
        // Create new pool
        game.pool.create('vectors');

        var vector = new game.Vector();

        // Put object into pool
        game.pool.put('vectors', vector);

        // Get object from pool
        var newVector = game.pool.get('vectors');
    }
});

game.start();

});
