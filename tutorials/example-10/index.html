Keyboard

game.module(
    'game.main'
)
.require(
    'engine.core',
    // Keyboard module not included in core
    'engine.keyboard'
)
.body(function() {

SceneGame = game.Scene.extend({
    keydown: function(key) {
        if(key === 'A') console.log('You pressed key A');
    },

    keyup: function(key) {
        if(key === 'A') console.log('You released key A');
    },

    update: function() {
        if(game.keyboard.down('B')) console.log('You are pressing key B');
        this.super();
    }
});

// You can even add more keycodes if needed
game.keyboard.keys[144] = 'NUMLOCK';

game.start();

});
