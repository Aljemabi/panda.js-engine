Grouping tweens

// Option 1 - Using one tween

game.scene.addTween(sprite, {
    position: {x: 200},
    alpha: 0
}, 2, {onComplete: function() {
    // Tween completed
}}).start();

game.scene.stopTweens(sprite);

// Option 2 - Using TweenGroup

var group = new game.TweenGroup(2, function() {
    // Tween completed
});

group.add(sprite, {alpha: 0});
group.add(sprite.position, {x: 200});

group.start();

// Option 3 - Using array

var tweens = [];

var tween = game.scene.addTween(sprite.position, {x: 200}, 2, {onComplete: function() {
    // Tween completed
}});
tweens.push(tween);
var tween = game.scene.addTween(sprite.alpha, 0, 2);
tweens.push(tween);

for (var i = 0; i < tweens.length; i++) {
    tweens[i].start();
}
