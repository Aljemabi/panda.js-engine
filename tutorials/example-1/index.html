Tweening

game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function(){

game.addAsset('media/player.png');

SceneGame = game.Scene.extend({
    init: function() {
        var sprite = new game.Sprite(100, 100, 'media/player.png');
        this.stage.addChild(sprite);
        
        // Tween multiple properties
        game.scene.addTween(sprite, {
            position: {x: 200},
            alpha: 0
        }, 2, {onComplete: function() {
            // Tween complete
        }}).start();

        // Use TweenGroup
        var group = new game.TweenGroup(2, function() {
            // Tween complete
        });

        group.add(sprite.position, {y: 200});
        group.add(sprite.scale, {x: 2});

        group.start();
    }
});

game.start(SceneGame);

});
