Sprites

game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function(){

game.addAsset('media/player.png');

SceneGame = game.Scene.extend({
    init: function() {
        // Create sprite
        var sprite = new game.Sprite(100, 100, 'media/player.png');

        // Change sprite position
        sprite.position.x = 200;
        sprite.position.y = 400;

        // Change sprite scale
        sprite.scale.x = 0.5;
        sprite.scale.y = 2.0;

        // Change sprite alpha
        sprite.alpha = 0.5;

        // Change sprite rotation
        sprite.rotation = 0.2; // radians

        // Change sprite anchor
        sprite.anchor.x = 0.5;
        sprite.anchor.y = 1.0;

        // Change sprite texture
        sprite.setTexture(game.Texture.fromImage('media/player.png'));

        // Hide sprite
        sprite.visible = false;

        // Get sprite dimensions
        var width = sprite.width;
        var height = sprite.height;

        // Add sprite to stage
        this.stage.addChild(sprite);
    }
});

game.start(SceneGame);

});
