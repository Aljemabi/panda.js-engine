Sprite sheets

Create sprite sheets with TexturePacker using data format JSON (Hash)
game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function(){

// Load spritesheet json
game.addAsset('media/spritesheet.json');

SceneGame = game.Scene.extend({
    init: function() {
        // Load sprite from spritesheet
        var sprite = new game.Sprite(0, 0, 'logo1.png');
        this.stage.addChild(sprite);
    }
});

game.start();

});
