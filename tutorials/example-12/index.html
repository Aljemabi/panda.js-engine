Accelerometer

New in version 1.1
game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function() {

game.addAsset('media/logo.png');

SceneGame = game.Scene.extend({
    init: function() {
        this.sprite = new game.Sprite(0, 0, 'media/logo.png');
        this.stage.addChild(this.sprite);
    },

    update: function() {
        if(game.accel) {
            this.sprite.position.x += game.accel.x * 50 * game.system.delta;
            this.sprite.position.y += game.accel.y * 50 * game.system.delta;
        }
        this.super();
    }
});

game.start();

});
