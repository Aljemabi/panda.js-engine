Dynamic loader

New in version 1.1
game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function(){

SceneTitle = game.Scene.extend({
    init: function() {
        var sprite = new game.Sprite(0, 0, 'media/logo.png');
        this.stage.addChild(sprite);
    },

    click: function() {
        game.addAsset('media/player.png');
        game.addSound('media/sound/jump.m4a', 'jump');

        var loader = new game.Loader(SceneGame);
        loader.start();
    }
});

SceneGame = game.Scene.extend({
    backgroundColor: 0xff0000,

    init: function() {
        var sprite = new game.Sprite(100, 100, 'media/player.png');
        this.stage.addChild(sprite);

        game.sound.playSound('jump');
    }
});

game.addAsset('media/logo.png');

game.start(SceneTitle);

});
