Analytics

game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function() {
    SceneGame = game.Scene.extend({
        init: function() {
            // Send event to analytics
            game.analytics.event('start');
        },

        gameOver: function() {
            game.analytics.event('gameover');  
        }
    });

    // Set id for Google Analytics
    game.Analytics.id = 'UA-12345678-9':

    game.start(SceneGame);
});
