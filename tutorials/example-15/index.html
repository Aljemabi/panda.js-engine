Sprite animation

game.module(
    'game.main'
)
.require(
    'engine.core'
)
.body(function(){

// Load images for animation (you can also use sprite sheets)
game.addAsset('media/player1.png');
game.addAsset('media/player2.png');

SceneGame = game.Scene.extend({
    init: function() {
        // Create new sprite animation
        var anim = new game.MovieClip([
            game.Texture.fromImage('media/player1.png'),
            game.Texture.fromImage('media/player2.png')
        ]);

        // Set position
        anim.position.x = 100;
        anim.position.y = 100;

        // Set animation speed
        anim.animationSpeed = 0.1;

        // Play animation
        anim.play();

        // Add animation to stage
        this.stage.addChild(anim);
    }
});

game.start();

});
