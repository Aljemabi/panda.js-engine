Simple particles

game.module(
    'game.main'
)
.require(
    'engine.core',
    'engine.particle' // Particle engine is not required from core
)
.body(function(){

game.addAsset('media/particle.png');

SceneGame = game.Scene.extend({
    init: function() {
        var emitter = new game.Emitter();
        emitter.container = this.stage;
        emitter.textures.push('media/particle.png');
        // Put emitter on center of screen
        emitter.position.x = game.system.width / 2;
        emitter.position.y = game.system.height / 2;

        this.addEmitter(emitter);
    }
});

game.start(SceneGame);

});
